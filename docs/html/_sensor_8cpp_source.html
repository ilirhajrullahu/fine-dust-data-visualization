119 void Sensor::readAndExportRecordData(string fileName) {
120  ifstream fin;
121  string line;
122  // Open an existing file
123  fin.open(fileName, ifstream::in);
124  getline(fin, line); //ignore first line
125 
126  Connection c;
127 
128  qDebug() << "STARTED writing data into DB";
129 
130  while (getline(fin, line)) {
131  vector<string> raw = splitRawString(line, ';');
132  string timestamp = this->reformatTime(raw[5]);
133  if (sensor_type == "SDS011" || sensor_type == "PPD42NS") {
134  double p_p1 = std::stod(raw[6]);
135  double p_p2 = std::stod(raw[9]);
136  c.postData(this->sensor_id, this->sensor_type,
137  this->longitude, this->latitude, p_p1, p_p2, timestamp);
138  }
139  else { //DHT22
140  double temerature = std::stod(raw[6]);
141  double humidity = std::stod(raw[7]);
142  c.postData(this->sensor_id, this->sensor_type,
143  this->longitude, this->latitude, temerature, humidity, timestamp);
144  }
145 
146  }
147  fin.close();
148 
149  qDebug() << "STOPPED writing data into DB";
150 }
